import { IAddress, Message, Session, UniversalBot } from 'botbuilder';
import { ConversationState, IConversation } from '../IConversation';
import { MessageReceivedWhileWaitingHandler } from '../options/MessageReceivedWhileWaitingHandler';
import { IProvider } from './../provider/IProvider';
 
export function routeCustomerMessage(
    bot: UniversalBot,
    provider: IProvider,
    messageReceivedWhileWaitingHandler: MessageReceivedWhileWaitingHandler
): (s: Session, n: Function) => void {
    return (session: Session, next: Function) => {
        const customerAddress = session.message.address;
 
        return provider.getConversationFromCustomerAddress(customerAddress)
            .then((convo: IConversation) => {
                if (convo) {
                    const agentAddress = convo.agentAddress;
                    const agentMirrorMessage = Object.assign({}, session.message, { address: agentAddress, value: 'customerMessage'});
 
                    switch (convo.conversationState) {
                        case ConversationState.Watch:
                            bot.send(agentMirrorMessage);
 
                            return next();
                        case ConversationState.Bot:
                            return next();
                        case ConversationState.Agent:
                            return bot.send(agentMirrorMessage);
                        case ConversationState.Wait:
                            messageReceivedWhileWaitingHandler(bot, session, next);
                            break;
                        case ConversationState.WatchAndWait:
                            messageReceivedWhileWaitingHandler(bot, session, next);
                            bot.send(agentMirrorMessage);
                            break;
                        default:
                            return next();
                    }
                } else {
                    return provider.addCustomerMessageToTranscript(session.message)
                    //tslint:disable
                        .then(() => next());
                    //tslint:enable
                }
            });
    };
}
